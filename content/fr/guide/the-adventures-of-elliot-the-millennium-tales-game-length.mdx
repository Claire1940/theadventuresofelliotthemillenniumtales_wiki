<Steps>
  <Step num="1" title="Prioriser les points de repère">

  <Step num="2" title="Utiliser Faie pour la reconnaissance">

  <Step num="3" title="Se concentrer sur les synergies d'armes">

  <Step num="4" title="Récupérer les récompenses tôt">
